Aliases pour Bash

# Emacs
alias emacs='emacs -nw'

# Git
alias gd='git diff'
alias gl='git log --decorate --graph --all --oneline -15'
alias gll='git log --decorate --graph --all --oneline'
alias gs='git status'

# Unix commands
alias l='ls -CF'
alias la='ls -A'
alias ll='ls -alF'
alias ls='ls --color=auto'
alias md='mkdir'
alias del='rm *~ \#* 2&> /dev/null || true'
alias grep='grep --color'

# Debug / profiling
alias memreport='ms_print'
alias memusage='valgrind --trace-children=yes --tool=massif'

# Python
alias virtenv='python3 -m venv . && source ./bin/activate'

# Others
alias xclip='xclip -selection c'
alias brightness='xrandr --output eDP-1 --brightness'
